Les filtres

grayscale : niveau de gris


.filter-grayscale {
	filter: grayscale(1);
	-webkit-filter: grayscale(1);
}

sepia : effet sépia


.filter-sepia {
	filter: sepia(1);
	-webkit-filter: sepia(1);
}

blur : effet de flou


.filter-blur {
	filter: blur(10px);
	-webkit-filter: blur(10px);
}

drop-shadow : ombre portée


.filter-drop-shadow {
	filter: drop-shadow(5px 5px 10 black);
	-webkit-filter: drop-shadow(5px 5px 10 black);
}

saturate : saturation


.filter-saturate {
	filter: saturate(600%);
	-webkit-filter: saturate(600%);
}

hue-rotate : teinte


.filter-hue-rotate {
	filter: hue-rotate(150deg);
	-webkit-filter: hue-rotate(150deg);
}

invert : effet négatif


.filter-invert {
	filter: invert(1);
	-webkit-filter: invert(1);
}

opacity : opacité


.filter-opacity {
	filter: opacity(10%);
	-webkit-filter: opacity(10%);
}

brightness : luminosité


.filter-brightness {
	filter: brightness(50%);
	-webkit-filter: brightness(50%);
}

contrast : contraste


.filter-contrast {
	filter: contrast(50%);
	-webkit-filter: contrast(50%);
}
